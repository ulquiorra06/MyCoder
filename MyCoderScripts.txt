CREATE TABLE Users
(
	[UserId] TINYINT PRIMARY KEY IDENTITY, 
	[EmailId] VARCHAR(50) unique,
	[UserPassword] VARCHAR(15) NOT NULL,
	[Role] VARCHAR(50) not null,
	
)
GO


CREATE TABLE Project
(
	[ProjectId] CHAR(4)  PRIMARY KEY identity(100,1) ,
	[ProjectName] VARCHAR(50)  UNIQUE NOT NULL,
	[ProjDescription] VARCHAR(500) NOT NULL,
	[BuyerId] TINYINT  REFERENCES Buyer(BuyerId),
	[TechnicalDescription] VARCHAR(50)  NOT NULL,
	[Duration] TIMESPAN NOT NULL,
	[Status] VARCHAR(15) NOT NULL
)
GO



CREATE TABLE PaymentDetails
(
	[PaymentId] BIGINT PRIMARY KEY IDENTITY(1000,1),
	[EmailId] VARCHAR(50) REFERENCES Users(EmailId),
	[ProjectId] CHAR(4) REFERENCES ProjectDescription(ProjectId),
	[DateOfPurchase] SMALLDATETIME CHECK(DateOfPurchase<=GETDATE()) DEFAULT GETDATE() NOT NULL,
)
GO

CREATE TABLE Coder
(
	[CoderId] TINYINT PRIMARY KEY IDENTITY,
	[UserID] TINYINT REFERENCES Users(UserId),
	[ProjectId] CHAR(4) REFERENCES ProjectDescription(ProjectId),
	[BuyerId] TINYINT  REFERENCES Buyer(BuyerId),
	[PhoneNumber] NUMERIC(10),
	[LanguageProficiency] VARCHAR(50)  NOT NULL,
	[Country] VARCHAR(50) NOT NULL
)

CREATE TABLE Buyer
(
	[BuyerId] TINYINT PRIMARY KEY IDENTITY,
	[UserID] TINYINT REFERENCES Users(UserId),
	[BuyerName] VARCHAR(30),
	[PhoneNumber] NUMERIC(10),
	[Country] VARCHAR(50) NOT NULL
)



CREATE TABLE ProjectFeedback
(
	[ProjectId] CHAR(4) references ProjectDescription(ProjectId) ,
	[ProjectName] VARCHAR(50) references ProjectDescription(ProjectName),
	[CoderId] TINYINT  REFERENCES Coder(CoderId),
	[Feedback] VARCHAR(500)
	
)

CREATE TABLE Enroll
(
	[ProjectId] CHAR(4) references ProjectDescription(ProjectId) ,
	[ProjectName] VARCHAR(50) references ProjectDescription(ProjectName),
	[CoderId] TINYINT  REFERENCES Coder(CoderId)	
)

CREATE TABLE Rating
(
	
	[UserId] TININT references Users(UserId),
	[Rating] TINYINT CHECK(Rating IN (1,10)),
	[NoOfRatings] BIGINT	
)

CREATE TABLE Auction
(
	AuctionId TINYINT primary key,
	CoderId TINYINT REFERENCES Coder(CoderId), 
	NoOfBuyers SMALLINT ;
	BidHolder TINYINT  REFERENCES Buyer(BuyerId);
	FinalPrice BIGINT;
)

create procedure usp_calculate_rating
{ 
	@new_rating int,
	@coder_id int
}
as
begin
	declare @rating int
	declare @count int
	select  @rating=Rating,@count=NoOfRatings from Rating where Ratingid=@coder_id
	set @rating=(@count*@rating+@new_rating)/(@count+1)
	update Rating set rating=@rating,NoOfRatings=@count+1 where Ratingid=@coder_id
end
